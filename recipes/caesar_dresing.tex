\begin{recipe}
	[ %
	preparationtime = {\unit[5]{min}} ,
	%bakingtime={\unit[30]{min}} ,
	%bakingtemperature={\protect\bakingtemperature{fanoven=\unit[230]{?C}}} ,
	%portion = {\portion[porce]{4}},
	portion = {\portion{4}},
	%calory={\unit[3]{kJ}} ,
	source = {Roman Vaněk}
	]{Caesar dresink}

	%description

	\ingredients{
		100 g & majonézy\\
		1 lžička & kremžeké hořčice\\
		1 lžička & hrubozrné hořčice\\
		10 g & najemno nastrouhaného parmezánu\\
		1,5 & ančovičky\\
		1 stroužek & česneku\\
		1 kávová lžička & citronové šťávy\\
	}

	\preparation{
		\step Česnek podélně rozpůlíme, odstraníme vnitřní klíček a česnek nastrouháme na jemném struhadle nebo utřeme se špetkou soli na pastu.
		\step Ančovičky nakrájíme najemno a smícháme je s česnekem, ostatními surovinami a 3 lžícemi vody.
	}
\end{recipe}
