\begin{recipe}
	[ %
	preparationtime = {\unit[20]{min}} ,
	bakingtime={\unit[30]{min}} ,
	%bakingtemperature={\protect\bakingtemperature{fanoven=\unit[230]{?C}}} ,
	%portion = {\portion[porce]{4}},
	portion = {\portion{2}},
	%calory={\unit[3]{kJ}} ,
	source = {\href{https://www.indianhealthyrecipes.com/spicy-andhra-egg-curry-south-indian-egg-recipes/#wprm-recipe-container-38868}{IndianHealthyRecipes}}
	]{Aniččino oblíbené jihoindické egg-curry}

	%description

	\ingredients{
		4 & vejce natvrdo \\
		2 lžíce & oleje \\
		1/2 až 3/4 lžičky & kmínu \\
		& kari \\
		1 hrnek & cibule \\
		1 & zelená chilli paprička \\
		1 a 1/4 lžičky & zázvorovo-česneková pasty (poměr 1:1) \\
		2 & oloupaná rajčata \\
		8 & kešu oříšků \\
		1/8 lžičky & kurkumy \\
		3/4 až 1 lžička & garam masala v prášku \\
		1 lžička & koriandru v prášku\\
		& sůl \\
		3 lžíce & listů koriandru \\
		3 lžíce & kokosového mléka \\
	}

	\preparation{
		\step Oloupeme uvařená vejce a částečně je nakrojíme.
		\step Rajčata a kešu oříšky rozmixujeme a vytvoříme z nich pyré.
		\step Lehce osmažíme na oleji kmín (v kuse). Poté přidáme kari a po chvilce přisypeme cibuli a zelenou chilli papričku. Smažíme, dokud cibule nezačne hnědnout.
		\step Přidáme zázvorovo-česnekovou pastu a smažíme, dokud nezmizí výrazná vůně česneku.
		\step Přilijeme rajčatové pyré a smažíme dokud se rajčata a cibule nepromísí.
		\step Přisypeme červený chilli prášek, kurkumu, sůl, garam masalu a koriandr v prášku. Dobře promícháme a smažíme pár minut, dokud nezmizí surová vůně koření.
		\step Přidáme vejce a smažíme 2 minuty.
		\step Přilijeme tak akorát vody, aby ze směsy vznikla řidší omáčka. Vaříme pod pokličkou na středním ohni. Po pár minutách omáčka zhoustne.
		\step Přilijeme kokosové mléku a necháme projít varem. Pozor, kokosové mléko nevaříme příliš dlouho.
		\step Posypeme lístky koriandru a můžeme podávat.
	}
\end{recipe}
