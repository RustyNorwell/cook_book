\begin{recipe}
	[ %
	preparationtime = {\unit[20]{min}} ,
	bakingtime={\unit[5-7]{h}} ,
	%bakingtemperature={\protect\bakingtemperature{fanoven=\unit[230]{?C}}} ,
	%portion = {\portion[porce]{4}},
	portion = {\portion{4}},
	%calory={\unit[3]{kJ}} ,
	source = {Adam Janata}
	]{Guláš}

	%description

	\ingredients{
		800 g & hovězího\\
		1 kg & cibule\\
		 & uzená paprika\\
		 & sladká paprika\\
		 & pálivá paprika\\
		 & pepř\\
		 & majoránka\\
		 & kmín\\
		kostka & masoxu\\
		 & sůl\\
	}

	\preparation{
		\step Na másle osmažíme nakrájenou cibuli.
		\step Poté, co cibule lehce zhnědne přidáme na kostičky nakrájené maso a lehce maso osmažíme.
		\step Směs zalijeme vařící vodou.
		\step Přidáme cca jeden a půl pytlíku (velikost standardního Vitana pytlíku s kořením) uzené papriky, trochu sladký papriky, trochu pálivý papriky, dost pepře, hodně majoránky, dost kmínu a kostku nebo dvě kostky masoxu. Případně dle potřeby osolíme.
		\step Vaříme cca 5 - 7 hodin, dokud se maso nerozpadne na vlákna. V případě, že se guláš připaluje, podlijeme vodou. Pokud je moc hustý taktéž dokompenzujeme vodou.
	}
\end{recipe}
