\begin{recipe}
	[ %
	preparationtime = {\unit[10]{min}} ,
	bakingtime={\unit[8-9]{h}} ,
	bakingtemperature={\protect\bakingtemperature{fanoven=\unit[100]{\textcelsius}}},
	portion = {\portion[bochník]{1}},
	source = {Hveragerdi muzeum (Island)}
	]{Rugbraud}

	\ingredients{
		320 g & žitné mouky\\
		85 g & hladké pšeničné mouky\\
		200 g & cukru\\
		8 g & prášku na pečení\\
		1 g & soli\\
		500 ml & mléka\\
	}

	\preparation{
		\step Předehřejeme troubu na 100 \textcelsius.
		\step Smícháme všechny suché ingredience ve středně velké míse.
		\step Přidáme mléko a promícháme. Těsto bude poměrně řídké.
		\step Vytřeme kovový pekáč s výkem. Pokud nemáme víko, lze ho nahradit hliníkovou folií.
		\step Pečeme v troubě při 100 \textcelsius\ po dobu 8-9 hodin, dokud chléb není uprostřed upečený.
		\step Vyklepneme chlép do útěrky a zabalíme ho do ní. Necháme ho vychladnout, aby se na něm nevytvořila tvrdá krusta.
		\step Podáváme tence krájený s velkou vrstvou másla.
	}
\end{recipe}
